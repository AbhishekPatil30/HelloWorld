pipeline {
    agent any
    tools{
        maven 'local_maven'
    }
    stages {
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
            post{
                success{
                    echo "Archiving the Artifacts"
                    archiveArtifacts artifacts: '**/target/*.war'
                }
            }
        }
stage ('Deploy to tomcat') {
            steps{
                sshagent(['tomcat']) {
                  sh "scp -o stricthostkeychecking=no webapp/target/webapp.war ec2-user@3.111.214.108:/opt/tomcat/apache-tomcat-9.0.87/webapps"
}
            }
        }
    }
}
